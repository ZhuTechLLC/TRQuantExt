### 韬睿平台周转率因子计算

<div class="board-details">
  <div class="board-card">

```python
def calc_turnover_factors(stocks, date):
    """
    计算周转率因子
    """
    df = get_fundamentals(query(
        valuation.code,
        indicator.total_asset_turnover,
        indicator.inventory_turnover,
        indicator.receivable_turnover
    ).filter(valuation.code.in_(stocks)), date=date)
    
    return df

def calc_turnover_change(stocks, date):
    """
    计算周转率变化
    """
    current = calc_turnover_factors(stocks, date)
    last_year = date - pd.DateOffset(years=1)
    previous = calc_turnover_factors(stocks, last_year)
    
    df = current.merge(previous, on='code', suffixes=('_current', '_prev'))
    df['turnover_change'] = (
        df['total_asset_turnover_current'] - 
        df['total_asset_turnover_prev']
    )
    
    return df[['code', 'turnover_change']]
```
