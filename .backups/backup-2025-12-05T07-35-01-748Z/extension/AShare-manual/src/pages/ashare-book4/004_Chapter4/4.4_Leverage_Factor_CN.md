### 韬睿平台杠杆因子计算

  <div class="board-card">
    <h4>📊 杠杆因子计算代码</h4>

```python
def calc_leverage_factors(stocks, date):
    """
    计算杠杆因子
    """
    df = get_fundamentals(query(
        valuation.code,
        balance.total_liability / balance.total_assets,  # 资产负债率
        indicator.debt_to_asset_ratio,
        indicator.equity_multiplier
    ).filter(valuation.code.in_(stocks)), date=date)
    
    # 计算低杠杆因子（负债率取反）
    df['low_leverage'] = -df['debt_to_asset_ratio']
    
    return df

def low_leverage_quality_strategy(date, hold_num=30):
    """
    低杠杆高质量策略
    
    筛选条件：
    1. 资产负债率 < 60%
    2. ROE > 15%
    3. 毛利率 > 30%
    """
    stocks = get_index_stocks('000300.XSHG', date)
    
    df = get_fundamentals(query(
        valuation.code,
        indicator.debt_to_asset_ratio,
        indicator.roe,
        indicator.gross_profit_margin
    ).filter(
        valuation.code.in_(stocks),
        indicator.debt_to_asset_ratio < 60,
        indicator.roe > 15,
        indicator.gross_profit_margin > 30
    ), date=date)
    
    # 按ROE排序
    df = df.sort_values('roe', ascending=False)
    
    return df.head(hold_num)['code'].tolist()
```
