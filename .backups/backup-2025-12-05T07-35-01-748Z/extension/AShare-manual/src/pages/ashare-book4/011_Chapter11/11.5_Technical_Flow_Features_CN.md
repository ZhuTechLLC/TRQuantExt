```python
# 韬睿量化平台量价因子
def volume_price_factors(stock, date):
    """
    计算量价因子
    """
    prices = get_price(stock, end_date=date, count=60)
    
    # 成交量变化率
    vol_change = prices['volume'].iloc[-5:].mean() / prices['volume'].iloc[-60:-5].mean()
    
    # 量价配合度
    price_up = prices['close'].diff() > 0
    vol_up = prices['volume'].diff() > 0
    vol_price_sync = (price_up == vol_up).mean()
    
    # 突破成交量
    vol_breakout = prices['volume'].iloc[-1] / prices['volume'].iloc[-20:].mean()
    
    return {
        'vol_change': vol_change,
        'vol_price_sync': vol_price_sync,
        'vol_breakout': vol_breakout
    }
```
