0. 版本对比分析

0.1 策略演进概述

AdaptiveMomentumStrategy v1.0.0是基于固定参数动量策略（OptimizedMomentumStrategy）的重大升级版本。核心创新在于引入了市场环境自动识别框架，能够根据当前市场阶段动态调整策略参数，实现真正的自适应交易。

主要改进目标：

  • 提升策略在不同市场环境下的适应性
  • 改善风险控制能力，降低最大回撤
  • 提高风险调整后收益（Sharpe Ratio）
  • 保持高收益的同时降低波动性

版本演进路线图：

  1. v0.x (OptimizedMomentumStrategy)：固定参数动量策略，在牛市表现优异但熊市回撤大
  2. v1.0.0 (AdaptiveMomentumStrategy)：引入市场环境识别，实现参数动态调整
  3. 未来版本规划：增加机器学习模块、多策略组合、实时风险监控

# region imports
from AlgorithmImports import *
from enum import Enum
# endregion

class MarketRegime(Enum):
    """市场阶段枚举"""
    # 牛市阶段
    FULL_BULL_MARKET = "FullBullMarket"              # 全面牛市
    STRUCTURAL_BULL_MARKET = "StructuralBullMarket" # 结构性牛市/后期
    GROWTH_BULL_MARKET = "GrowthBullMarket"         # 成长股活跃期
    
    # 熊市阶段
    PANIC_BEAR_MARKET = "PanicBearMarket"           # 熊市恐慌阶段
    PERSISTENT_BEAR_MARKET = "PersistentBearMarket" # 持续熊市
    BEAR_MARKET_BOTTOM = "BearMarketBottom"         # 熊市末期筑底
    
    # 震荡市
    NEUTRAL_SIDEWAYS = "NeutralSideways"            # 中性震荡市
    WAITING_BREAKOUT = "WaitingBreakout"           # 等待突破
    TRANSITION_PHASE = "TransitionPhase"           # 牛熊转换盘整期
    
    # 特殊阶段
    INDUSTRY_ROTATION = "IndustryRotation"         # 行业轮动市
    HIGH_GROWTH_ACTIVE = "HighGrowthActive"         # 高增长板块活跃期

class TrendState(Enum):
    """趋势状态"""
    BULLISH = "Bullish"
    BEARISH = "Bearish"
    SIDEWAYS = "Sideways"

class TrendStrength(Enum):
    """趋势强度"""
    TRENDING = "Trending"
    CONSOLIDATING = "Consolidating"

class RiskMode(Enum):
    """风险模式"""
    NORMAL = "Normal"
    ELEVATED = "Elevated"
    PANIC = "Panic"

class MarketRegimeDetector:
    """
    多因素综合的市场环境自动识别框架
    
    基于PDF文档《市场环境的阶段识别与交易策略适配》设计
    包含四个层次：
    1. 趋势判定层（长期方向）
    2. 宽度及风险层（市场广度和情绪）
    3. 风格轮动层（板块相对强弱）
    4. 宏观过滤层（利率、通胀、流动性）
    """
    
    def __init__(self, algorithm):
        """初始化市场环境识别器"""
        self.algorithm = algorithm
        
        # 趋势判定层指标
        self.spy_ma200 = None
        self.spy_adx = None
        
        # 宽度及风险层指标
        self.vix = None
        
        # 风格轮动层指标
        self.qqq_spy_ratio_value = None
        
        # 当前市场状态
        self.current_regime = None
        self.trend_state = TrendState.SIDEWAYS
        self.trend_strength = TrendStrength.CONSOLIDATING
        self.risk_mode = RiskMode.NORMAL
        self.style_trend_growth = False
        
    def Initialize(self):
        """初始化所有指标"""
        # 1. 趋势判定层：SPY和200日均线、ADX
        spy = self.algorithm.AddEquity("SPY", Resolution.Daily)
        self.spy_ma200 = self.algorithm.SMA("SPY", 200, Resolution.Daily)
        self.spy_adx = self.algorithm.ADX("SPY", 14, Resolution.Daily)
        
        # 2. 宽度及风险层：VIX指数
        try:
            vix = self.algorithm.AddIndex("VIX", Resolution.Daily)
            self.vix = self.algorithm.SMA("VIX", 5, Resolution.Daily)
        except:
            try:
                vxx = self.algorithm.AddEquity("VXX", Resolution.Daily)
                self.vix = self.algorithm.SMA("VXX", 5, Resolution.Daily)
            except:
                self.vix = None
        
        # 3. 风格轮动层：QQQ/SPY比率
        qqq = self.algorithm.AddEquity("QQQ", Resolution.Daily)
        self.qqq_spy_symbol = Symbol.Create("QQQ", SecurityType.Equity, Market.USA)
        self.spy_symbol = Symbol.Create("SPY", SecurityType.Equity, Market.USA)
    
    def Update(self):
        """更新所有指标并判断当前市场环境"""
        if self.algorithm.IsWarmingUp:
            return MarketRegime.NEUTRAL_SIDEWAYS
        
        self._UpdateQQQSPYRatio()
        trend_state, trend_strength = self._AnalyzeTrend()
        risk_mode = self._AnalyzeRisk()
        style_trend = self._AnalyzeStyle()
        
        self.trend_state = trend_state
        self.trend_strength = trend_strength
        self.risk_mode = risk_mode
        self.style_trend_growth = style_trend
        
        regime = self._MakeDecision()
        self.current_regime = regime
        return regime
    
    def _UpdateQQQSPYRatio(self):
        """更新QQQ/SPY比率"""
        try:
            if (self.qqq_spy_symbol in self.algorithm.Securities and 
                self.spy_symbol in self.algorithm.Securities):
                qqq_price = self.algorithm.Securities[self.qqq_spy_symbol].Close
                spy_price = self.algorithm.Securities[self.spy_symbol].Close
                if spy_price > 0:
                    self.qqq_spy_ratio_value = qqq_price / spy_price
        except:
            self.qqq_spy_ratio_value = None
    
    def _AnalyzeTrend(self):
        """分析趋势状态和强度"""
        if not self.spy_ma200.IsReady or not self.spy_adx.IsReady:
            return TrendState.SIDEWAYS, TrendStrength.CONSOLIDATING
        
        spy_price = self.algorithm.Securities["SPY"].Close
        ma200 = self.spy_ma200.Current.Value
        adx = self.spy_adx.Current.Value
        
        if spy_price > ma200:
            trend_state = TrendState.BULLISH
        elif spy_price < ma200:
            trend_state = TrendState.BEARISH
        else:
            trend_state = TrendState.SIDEWAYS
        
        if adx > 25:
            trend_strength = TrendStrength.TRENDING
        else:
            trend_strength = TrendStrength.CONSOLIDATING
        
        return trend_state, trend_strength
    
    def _AnalyzeRisk(self):
        """分析风险模式"""
        if self.vix is None or not self.vix.IsReady:
            return RiskMode.NORMAL
        
        vix_value = self.vix.Current.Value
        if vix_value > 30:
            return RiskMode.PANIC
        elif vix_value > 20:
            return RiskMode.ELEVATED
        else:
            return RiskMode.NORMAL
    
    def _AnalyzeStyle(self):
        """分析风格趋势（成长 vs 价值）"""
        if self.qqq_spy_ratio_value is None:
            return False
        
        ratio = self.qqq_spy_ratio_value
        if hasattr(self, 'qqq_spy_history'):
            self.qqq_spy_history.append(ratio)
            if len(self.qqq_spy_history) > 20:
                self.qqq_spy_history.pop(0)
            avg_ratio = sum(self.qqq_spy_history) / len(self.qqq_spy_history)
            return ratio > avg_ratio
        else:
            self.qqq_spy_history = [ratio]
            return ratio > 1.0
    
    def _MakeDecision(self):
        """综合决策，确定最终市场阶段"""
        # 决策树逻辑
        if self.trend_state == TrendState.BULLISH and self.risk_mode != RiskMode.PANIC:
            if (self.style_trend_growth and 
                self.trend_strength == TrendStrength.TRENDING):
                return MarketRegime.FULL_BULL_MARKET
            elif self.style_trend_growth and self.risk_mode == RiskMode.ELEVATED:
                return MarketRegime.HIGH_GROWTH_ACTIVE
            elif not self.style_trend_growth:
                return MarketRegime.STRUCTURAL_BULL_MARKET
            else:
                return MarketRegime.GROWTH_BULL_MARKET
        
        elif self.trend_state == TrendState.BEARISH:
            if self.risk_mode == RiskMode.PANIC:
                return MarketRegime.PANIC_BEAR_MARKET
            elif self.risk_mode == RiskMode.ELEVATED:
                return MarketRegime.PERSISTENT_BEAR_MARKET
            else:
                if self.vix is not None and self.vix.IsReady:
                    vix_value = self.vix.Current.Value
                    if vix_value < 20:
                        return MarketRegime.BEAR_MARKET_BOTTOM
                return MarketRegime.PERSISTENT_BEAR_MARKET
        
        elif self.trend_state == TrendState.SIDEWAYS:
            if self.trend_strength == TrendStrength.CONSOLIDATING:
                return MarketRegime.NEUTRAL_SIDEWAYS
            else:
                return MarketRegime.TRANSITION_PHASE
        
        return MarketRegime.NEUTRAL_SIDEWAYS
    
    def GetStrategyRecommendation(self):
        """根据当前市场阶段返回策略建议"""
        regime = self.current_regime
        recommendations = {
            MarketRegime.FULL_BULL_MARKET: {
                "strategy_type": "TrendFollowing",
                "position_size": 1.0,
                "description": "全面牛市：趋势跟随策略，满仓做多成长股"
            },
            MarketRegime.PANIC_BEAR_MARKET: {
                "strategy_type": "Defensive",
                "position_size": 0.1,
                "description": "熊市恐慌阶段：大幅减仓，持有现金或防御资产"
            },
            # ... 其他市场阶段的建议
        }
        return recommendations.get(regime, recommendations[MarketRegime.NEUTRAL_SIDEWAYS])

核心功能：

  • 趋势判定层：使用SPY vs 200日均线判断趋势方向，ADX判断趋势强度
  • 风险判定层：使用VIX指数判断市场恐慌程度（Normal/Elevated/Panic）
  • 风格轮动层：使用QQQ/SPY比率判断成长vs价值风格
  • 综合决策：通过决策树将市场环境分类为11种市场阶段

2. 策略背景与理论依据

2.1 策略原理

理论基础

本策略结合了动量效应理论和市场环境识别理论，主要理论支撑包括：

  • Jegadeesh-Titman动量效应（1993）：中期动量（3-12个月）在股票市场中持续存在，买入近期表现好的股票可以获得显著的超额收益。
  • 市场环境识别理论：不同市场环境下，相同的策略参数会产生不同的效果。通过识别市场环境并动态调整参数，可以提升策略表现。
  • 行为金融学：投资者对信息的反应存在延迟和过度反应，导致价格趋势延续，为动量策略提供盈利空间。
  • 自适应系统理论：能够根据环境变化自动调整的系统具有更强的鲁棒性和适应性。

市场假设

  • 市场环境可识别：通过技术指标（SPY vs MA200、ADX、VIX、QQQ/SPY比率）可以准确识别当前市场阶段
  • 参数适应性：不同市场环境下，最优策略参数存在显著差异
  • 动量延续性：在特定市场环境下（如全面牛市），价格趋势具有延续性
  • 风险可控性：通过市场环境识别，可以在高风险环境（如恐慌熊市）提前降低仓位或清仓

3. 策略实现详解

3.1 代码架构

本策略采用模块化设计，分为两个核心文件：

  • main.py（423行）：主策略文件，包含策略初始化、市场环境更新、选股、再平衡和持仓管理
  • strategies/market_regime_detector.py（421行）：市场环境识别模块，封装多因素市场环境识别逻辑

整体设计思路

策略采用事件驱动架构，通过QuantConnect的调度系统实现：

  • 每日收盘前：更新市场环境识别，动态调整策略参数
  • 每周一、三、五开盘后：执行选股和再平衡
  • 每日数据更新：检查止盈止损条件

模块划分

  • 市场环境识别模块：独立的MarketRegimeDetector类，可复用
  • 策略执行模块：AdaptiveMomentumStrategy类，继承QCAlgorithm
  • 参数调整模块：_AdjustStrategyParameters()方法，根据市场环境动态调整

关键数据结构

  • self.regime_detector：MarketRegimeDetector实例，负责市场环境识别
  • self.current_regime：当前市场阶段（MarketRegime枚举）
  • self.security_data：字典，存储每只股票的技术指标对象
  • self.positions：字典，存储当前持仓信息（entry价格、stop价格）

3.2 核心算法

市场环境识别算法

策略的核心创新在于多因素市场环境识别框架，包含三个层次：

  1. 趋势判定层：
    • SPY价格 vs 200日均线：判断趋势方向（Bullish/Bearish/Sideways）
    • ADX指标：判断趋势强度（Trending/Consolidating，阈值25）
  2. 风险判定层：
    • VIX指数：判断市场恐慌程度
    • VIX < 20：Normal风险
    • 20 ≤ VIX < 30：Elevated风险
    • VIX ≥ 30：Panic风险（触发恐慌熊市阶段）
  3. 风格轮动层：
    • QQQ/SPY比率：判断成长vs价值风格
    • 比率高于历史平均：成长风格占优
    • 比率低于历史平均：价值风格占优

通过决策树将上述三个层次的判断综合，确定11种市场阶段之一。

选股逻辑

选股采用多因子筛选 + 动态评分的方式：

  1. 初步筛选条件（根据市场环境动态调整）：
    • ROC10 ≥ 动态门槛（1.5%-5%，根据市场环境）
    • ROC20 ≥ 动态门槛（2.5%-8%，根据市场环境）
    • 成交量 ≥ 50日均量的0.8倍
    • RSI 在 25-95 之间
    • 价格条件：牛市需在SMA50以上，熊市末期允许5%折价
  2. 综合评分公式（权重根据市场环境调整）：
    • 全面牛市/高增长活跃期：Score = 0.6×ROC10 + 0.3×ROC20 + 0.1×VolumeRatio（重视短期动量）
    • 熊市末期：Score = 0.4×ROC10 + 0.4×ROC20 + 0.2×VolumeRatio（重视超跌反弹）
    • 其他环境：Score = 0.5×ROC20 + 0.35×ROC10 + 0.15×VolumeRatio（平衡各因子）
  3. 选择Top N：根据市场环境选择2-8只股票

择时逻辑

策略采用定期再平衡的方式，不依赖择时信号：

  • 再平衡频率：每周一、三、五（每周3次）
  • 市场环境变化时：自动调整参数，但不会立即再平衡
  • 恐慌熊市：自动清仓，等待市场环境改善

仓位管理

仓位管理采用等权重分配 + 动态调整的方式：

  • 单只股票仓位：根据市场环境动态调整（5%-15%）
  • 最大持仓数：根据市场环境动态调整（2-8只）
  • 总仓位：根据市场环境自动调整（恐慌熊市：0%，全面牛市：105%）

风险控制机制

风险控制采用多层次机制：

  1. 市场环境识别：在恐慌熊市自动清仓
  2. 动态止损：根据市场环境调整止损比例（5%-12%）
  3. 动态止盈：根据市场环境调整止盈比例（20%-70%）
  4. 仓位控制：在风险环境降低仓位和持仓数

策略的优势在于参数动态调整，能够根据市场环境自动优化参数组合，避免了固定参数在不同市场环境下的不适应问题。

参数优化过程

参数优化采用基于市场环境的参数映射方式：

  • 通过历史回测和理论分析，为每种市场阶段确定最优参数范围
  • 在回测中验证参数映射的有效性
  • 根据回测结果微调参数映射表
  • 策略在运行时根据识别的市场环境自动应用对应参数

这种方式避免了全局参数优化可能导致的过拟合问题，同时保证了策略在不同市场环境下的适应性。

    def Screen(self):
        """筛选与再平衡"""
        if self.IsWarmingUp:
            return
        
        # 如果市场环境不支持做多，跳过选股
        if self.current_regime in [MarketRegime.PANIC_BEAR_MARKET]:
            # 恐慌熊市：清仓观望
            for symbol in list(self.positions.keys()):
                self.Liquidate(symbol)
                del self.positions[symbol]
            return
        
        candidates = []
        
        for symbol, data in self.security_data.items():
            indicators = data["indicators"]
            if not all(ind.IsReady for ind in indicators.values()):
                continue
            
            if symbol not in self.Securities:
                continue
            
            price = self.Securities[symbol].Close
            volume = self.Securities[symbol].Volume
            
            roc_10 = indicators["roc_10"].Current.Value
            roc_20 = indicators["roc_20"].Current.Value
            volume_sma = indicators["volume_sma"].Current.Value
            rsi = indicators["rsi"].Current.Value
            sma_20 = indicators["sma_20"].Current.Value
            sma_50 = indicators["sma_50"].Current.Value
            
            # 根据市场环境调整筛选条件
            if roc_10 < self.roc_10_min:
                continue
            if roc_20 < self.roc_20_min:
                continue
            if volume_sma <= 0 or volume / volume_sma < self.volume_min:
                continue
            if not (self.rsi_min < rsi < self.rsi_max):
                continue
            
            # 不同市场环境的均线条件
            if self.current_regime in [MarketRegime.FULL_BULL_MARKET, 
                                      MarketRegime.GROWTH_BULL_MARKET,
                                      MarketRegime.HIGH_GROWTH_ACTIVE]:
                # 牛市：只需在SMA50以上
                if price < sma_50:
                    continue
            elif self.current_regime == MarketRegime.BEAR_MARKET_BOTTOM:
                # 熊市末期：可以买入超跌股（允许在均线下方）
                if price < sma_50 * 0.95:  # 允许5%的折价
                    continue
            else:
                # 其他情况：需要突破均线
                if price < sma_50:
                    continue
            
            # 打分（根据市场环境调整权重）
            volume_ratio = volume / volume_sma
            
            if self.current_regime in [MarketRegime.FULL_BULL_MARKET,
                                      MarketRegime.HIGH_GROWTH_ACTIVE]:
                # 牛市：更重视短期动量
                score = 0.6 * roc_10 + 0.3 * roc_20 + 0.1 * volume_ratio
            elif self.current_regime == MarketRegime.BEAR_MARKET_BOTTOM:
                # 熊市末期：重视超跌反弹
                score = 0.4 * roc_10 + 0.4 * roc_20 + 0.2 * volume_ratio
            else:
                # 默认：平衡各种因子
                score = 0.5 * roc_20 + 0.35 * roc_10 + 0.15 * volume_ratio
            
            candidates.append({
                "symbol": symbol,
                "ticker": data["ticker"],
                "score": score,
                "roc_10": roc_10
            })
        
        # 选择Top N（根据市场环境调整）
        candidates.sort(key=lambda x: x["score"], reverse=True)
        target_count = min(self.max_positions, len(candidates))
        
        # 清仓不在目标列表的
        target_symbols = {c["symbol"] for c in candidates[:target_count]}
        for symbol in list(self.positions.keys()):
            if symbol not in target_symbols:
                self.Liquidate(symbol)
                del self.positions[symbol]
        
        # 买入目标列表
        for c in candidates[:target_count]:
            if c["symbol"] not in self.positions:
                self.SetHoldings(c["symbol"], self.position_size)
                self.positions[c["symbol"]] = {
                    "entry": self.Securities[c["symbol"]].Close,
                    "stop": self.Securities[c["symbol"]].Close * (1 - self.stop_loss)
                }
                self.Log(f"买入: {c['ticker']}, 得分{c['score']:.3f}, 市场环境: {self.current_regime.value}")
    
    def OnData(self, data):
        """管理持仓"""
        if self.IsWarmingUp:
            return
        
        for symbol in list(self.positions.keys()):
            if symbol in data:
                bar = data[symbol]
                if bar is None:
                    continue
                    
                p = self.positions[symbol]
                price = bar.Close
                
                if price == 0:
                    continue
                
                ret = (price - p["entry"]) / p["entry"]
                
                # 止盈
                if ret >= self.take_profit:
                    self.Liquidate(symbol)
                    self.Log(f"止盈: {self.security_data[symbol]['ticker']}, 收益{ret*100:.1f}%")
                    del self.positions[symbol]
                    continue
                
                # 止损
                if price <= p["stop"]:
                    self.Liquidate(symbol)
                    self.Log(f"止损: {self.security_data[symbol]['ticker']}, 收益{ret*100:.1f}%")
                    del self.positions[symbol]
    
    def OnEndOfAlgorithm(self):
        """结束"""
        for symbol in list(self.positions.keys()):
            self.Liquidate(symbol)
