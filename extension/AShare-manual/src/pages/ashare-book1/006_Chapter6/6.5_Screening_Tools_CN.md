### Python量化筛选示例

<div class="tip-box">
  <h4>📊 使用JQData进行量化筛选</h4>
  <pre><code>import jqdatasdk as jq
import pandas as pd

jq.auth('账号', '密码')

# 定义筛选条件
query = jq.query(
    jq.valuation.code,
    jq.valuation.pe_ratio,
    jq.valuation.pb_ratio,
    jq.indicator.roe,
    jq.indicator.inc_revenue_year_on_year
).filter(
    jq.indicator.roe > 15,  # ROE > 15%
    jq.indicator.inc_revenue_year_on_year > 20,  # 收入增速 > 20%
    jq.valuation.pe_ratio < 30,  # PE < 30
    jq.valuation.pe_ratio > 0  # 排除亏损
)

# 执行筛选
df = jq.get_fundamentals(query, date='2024-01-01')

# 计算PEG
df['peg'] = df['pe_ratio'] / df['inc_revenue_year_on_year']

# 筛选PEG < 1的股票
result = df[df['peg'] < 1]
print(f"筛选结果：{len(result)}只股票")
</code></pre>
</div>
