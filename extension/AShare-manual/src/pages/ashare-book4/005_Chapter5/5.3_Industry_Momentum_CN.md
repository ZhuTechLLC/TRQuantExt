```python
def calc_industry_momentum(date, window=20):
    """
    计算行业动量因子
    
    参数:
        date: 计算日期
        window: 回溯窗口（天）
    
    返回:
        行业动量DataFrame
    """
    # 获取申万一级行业列表
    industries = get_industries('sw_l1')
    
    industry_momentum = []
    for ind_code in industries.index:
        # 获取行业指数收益
        ind_price = get_price(
            ind_code,
            end_date=date,
            count=window + 1,
            fields=['close']
        )
        
        # 计算动量
        momentum = ind_price['close'].iloc[-1] / ind_price['close'].iloc[0] - 1
        
        industry_momentum.append({
            'industry_code': ind_code,
            'industry_name': industries.loc[ind_code, 'name'],
            'momentum': momentum
        })
    
    df = pd.DataFrame(industry_momentum)
    
    # 计算动量排名
    df['momentum_rank'] = df['momentum'].rank(ascending=False)
    
    return df

def industry_rotation_strategy(date, top_n=3, hold_num=30):
    """
    行业轮动策略
    
    选择动量最强的N个行业，在每个行业中选择最强的股票
    """
    # 计算行业动量
    ind_momentum = calc_industry_momentum(date, window=20)
    
    # 选择动量最强的行业
    top_industries = ind_momentum.nlargest(top_n, 'momentum')['industry_code'].tolist()
    
    # 在强势行业中选股
    selected_stocks = []
    for ind_code in top_industries:
        ind_stocks = get_industry_stocks(ind_code, date)
        
        # 在行业内按动量排序
        stock_momentum = calc_stock_momentum(ind_stocks, date)
        top_stocks = stock_momentum.nlargest(hold_num // top_n, 'momentum')
        
        selected_stocks.extend(top_stocks['code'].tolist())
    
    return selected_stocks
```
