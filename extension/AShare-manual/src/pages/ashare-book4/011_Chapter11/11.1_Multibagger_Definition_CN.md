```python
# 韬睿量化平台高倍股筛选代码示例
def screen_multibagger(start_date, end_date, threshold=3.0):
    """
    筛选高倍股
    
    参数:
        start_date: 起始日期
        end_date: 结束日期
        threshold: 涨幅倍数阈值（默认3倍）
    
    返回:
        高倍股列表及其涨幅
    """
    # 获取所有股票
    all_stocks = get_all_securities()
    
    multibaggers = []
    for stock in all_stocks:
        # 获取期间最低价和最高价
        prices = get_price(stock, start_date, end_date)
        
        # 计算从最低点到最高点的涨幅
        min_price = prices['low'].min()
        max_price = prices['high'].max()
        
        # 确保最低点在最高点之前
        min_date = prices['low'].idxmin()
        max_date = prices['high'].idxmax()
        
        if max_date > min_date:
            gain = (max_price - min_price) / min_price
            if gain >= threshold - 1:  # 3倍 = 涨幅200%
                multibaggers.append({
                    'code': stock,
                    'gain': gain,
                    'start_price': min_price,
                    'end_price': max_price,
                    'start_date': min_date,
                    'end_date': max_date
                })
    
    return multibaggers
```
