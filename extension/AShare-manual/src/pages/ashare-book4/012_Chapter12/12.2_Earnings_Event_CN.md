```python
# 韬睿量化平台业绩超预期策略
def earnings_surprise_strategy(date):
    """
    业绩超预期策略
    """
    # 获取业绩预告数据
    forecast = get_STK_FIN_FORCAST(
        start_date=date,
        end_date=date
    )
    
    # 筛选业绩大幅增长
    positive = forecast[
        (forecast['change_type'] == '预增') |
        (forecast['change_type'] == '略增') |
        (forecast['change_type'] == '扭亏')
    ]
    
    # 筛选增长幅度
    positive = positive[
        positive['change_min'] > 30  # 增长下限 > 30%
    ]
    
    return positive['code'].tolist()
```
